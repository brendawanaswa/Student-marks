# Student-marks
#include<stdio.h>
int main()
{
char name[50];
float marks;
//Taking input from the user
printf("Enter Student name: ");
fgets(name; sizeof(name), stdin);
printf("Enter student marks: ");
scanf("%f", &marks);
//displaying the details
printf("\nStudent Details:\n");
printf("Name: %s", name);
printf("Marks: %2f\n", marks);
return 0;
}
